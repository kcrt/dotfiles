#!/usr/bin/env python3

a = 10
fruits = ["apple", "banana", "fig"]

# == if ==
if a < 10:
    print("a is less than 10.")
elif a == 10:
    print("a is equal to 10.")
else:
    print("a is greater than 10.")

# == 3項演算子 ==
print("a is negative" if a < 0 else "a is not negative")

# == for ==
for i in range(5):
    print(i)        # => 0, 1, 2, 3, 4

for food in fruits:
    print(food)

# == loop ==
while a > 0:
    print(a)
    a -= 1

# == break, continue ==
for i in range(10):
    if i == 4:
        continue    # 次のループへ
    elif i == 8:
        break       # このforを終わる
else:
    print("Yes!")   # ループを使い切ったときに実行(breakで抜けると実行されない)

# == switch(select case) ==
# Pythonにはswitchに相当する構文はない。
# if-elif-elif-...を駆使すること。
