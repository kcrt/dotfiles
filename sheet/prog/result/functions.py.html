#!/usr/bin/env python3


def myHello(greeting, name="kcrt"):
    return f"{greeting}, {name}."


# 型ヒント付きの時
def myHello2(greeting: str, name: str = "kcrt") -> str:
    return f"{greeting}, {name}."


myHello("Hello", "John")  # => Hello, John.
myHello("你好")   # => 你好, kcrt.
myHello(name="Mary", greeting="Good morning")  # => Good morning, Mary.
